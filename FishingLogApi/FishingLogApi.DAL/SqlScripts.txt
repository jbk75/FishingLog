CREATE VIEW V_TripWithFishingPlaceName AS
SELECT
    t.Id AS TripId,
    t.FishingPlaceId,
    fp.Name AS FishingPlaceName,
    t.DateFrom,
    t.DateTo,
    t.Description,
    t.DateCreated,
    t.LastModified
FROM
    Trip t
JOIN
    FishingPlace fp ON t.FishingPlaceId = fp.Id;

Go

    WindSpeedMetersPerSecond REAL,

);
